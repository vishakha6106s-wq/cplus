//Read two String user input and check if first contains second?
#include<iostream>
using namespace std;

int main()
{
    char a[100], b[50];
    int i, j, count;

    cout<<"Enter first string: ";
    cin>>a;

    cout<<"Enter second string: ";
    cin>>b;

    for(i=0; a[i]!='\0'; i++)
    {
        count = 0;

        for(j=0; b[j]!='\0'; j++)
        {
            if(a[i+j] == b[j])
                count++;
        }

        if(count > 0 && count == j)
        {
            cout<<"Yes, it contains";
            return 0;
        }
    }

    cout<<"No, it does not contain";

    return 0;
}/*OUTPUT
Enter first string: helllo
Enter second string: lo
Yes, it contains
*/
