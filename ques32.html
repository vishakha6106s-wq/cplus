Reverse String Word by Word


//Reverse String Word by Word
#include<iostream>
using namespace std;

int main()
{
    char str[50], word[10][10];
    int i=0, j=0, w=0;

    cout<<"Enter string: ";
    cin.getline(str,50);

    // store words
    while(str[i] != '\0')
    {
        if(str[i] != ' ')
        {
            word[w][j] = str[i];
            j++;
        }
        else
        {
            word[w][j] = '\0';
            w++;
            j = 0;
        }
        i++;
    }

    word[w][j] = '\0';  // last word

    // print reverse
    for(i=w; i>=0; i--)
    {
        cout<<word[i]<<" ";
    }

    return 0;
}/*OUTPUT
Enter string: hello bye Neha
Neha bye hello
