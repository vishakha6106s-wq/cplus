Implement a class Money to represent money. Overload the + operator to add two amounts of money.



#include <iostream>
using namespace std;

class Money {
    int rs, ps;

public:
    Money(int r=0, int p=0) {
        rs = r;
        ps = p;
    }

    Money operator+(Money m) {
        Money t;
        t.ps = ps + m.ps;
        t.rs = rs + m.rs + t.ps/100;
        t.ps = t.ps % 100;
        return t;
    }

    void show() {
        cout << rs << " Rs " << ps << " Ps";
    }
};

int main() {
    Money a(5,75), b(3,50), c;
    c = a + b;
    c.show();
}

Output


9 Rs 25 Ps
