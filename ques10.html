Program to count the total number of vowels and consonants in a string


#include<iostream>
using namespace std;

int main()
{
    char str[50];
    int i, v=0, c=0;

    cout<<"Enter string: ";
    cin>>str;

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u')
            v++;
        else
            c++;
    }

    cout<<"Vowels = "<<v<<endl;
    cout<<"Consonants = "<<c;

    return 0;
}

Output


Enter string: hello
Vowels = 2
Consonants = 3
