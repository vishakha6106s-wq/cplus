//Write a program to check if two Strings are created with same characters?
#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    char a[50], b[50];
    int i, j, found;

    cout<<"Enter first string: ";
    cin>>a;

    cout<<"Enter second string: ";
    cin>>b;

    if(strlen(a) != strlen(b))
    {
        cout<<"Not same characters";
        return 0;
    }

    for(i=0; a[i]!='\0'; i++)
    {
        found = 0;

        for(j=0; b[j]!='\0'; j++)
        {
            if(a[i] == b[j])
            {
                found = 1;
                break;
            }
        }

        if(found == 0)
        {
            cout<<"Not same characters";
            return 0;
        }
    }

    cout<<"Same characters";

    return 0;
}/*OUTTPUT
