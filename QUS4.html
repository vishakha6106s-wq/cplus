Program to print the duplicate elements of an array.


#include<iostream>
using namespace std;
int main()
{
int arr[50],n,i,j;
cout<<"Enter number of elements:";
cin>>n;
cout<<"Enter elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Duplicate elements are:";
for(i=0;i<n;i++)
{
for(j=i+1;j<n;j++)
{
if(arr[i]==arr[j])
{
cout<<arr[j]<<" ";
break;
}
}
}
return 0;
}
