Program to print the sum of all the items of the array.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,sum;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
sum=arr[0]+arr[1]+arr[2]+arr[3]+arr[4];
cout<<"Sum of all items="<<sum;
return 0;
}
