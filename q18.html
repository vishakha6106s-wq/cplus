Write a C++ program to create a class called "Restaurant" with attributes for menu items, prices, and ratings, and methods to add and remove items, and to calculate average rating.


#include <iostream>
#include <string>
using namespace std;

class Restaurant {
public:
    string items[5];   // max 5 items
    float prices[5];
    float ratings[5];
    int count;         // current number of items

    Restaurant() {
        count = 0;
    }

    void addItem(string name, float price, float rating) {
        if(count < 5) {
            items[count] = name;
            prices[count] = price;
            ratings[count] = rating;
            count++;
        } else {
            cout << "Menu full!\n";
        }
    }

    void removeItem(string name) {
        bool found = false;
        for(int i=0; i<count; i++) {
            if(items[i] == name) {
                for(int j=i; j<count-1; j++) {
                    items[j] = items[j+1];
                    prices[j] = prices[j+1];
                    ratings[j] = ratings[j+1];
                }
                count--;
                found = true;
                cout << name << " removed from menu.\n";
                break;
            }
        }
        if(!found) cout << name << " not found in menu.\n";
    }

    void showMenu() {
        cout << "Menu:\n";
        for(int i=0; i<count; i++) {
            cout << items[i] << " - Rs." << prices[i] << " - Rating: " << ratings[i] << endl;
        }
    }

    void averageRating() {
        if(count == 0) {
            cout << "No items in menu.\n";
            return;
        }
        float sum = 0;
        for(int i=0; i<count; i++) sum += ratings[i];
        cout << "Average rating = " << sum/count << endl;
    }
};

int main() {
    Restaurant r;

    // Add items
    r.addItem("Burger", 120, 4.5);
    r.addItem("Pizza", 250, 4.8);
    r.addItem("Pasta", 180, 4.2);

    // Show menu and average rating
    r.showMenu();
    r.averageRating();

    // Remove an item
    r.removeItem("Pizza");
    r.showMenu();
    r.averageRating();

    return 0;
}
