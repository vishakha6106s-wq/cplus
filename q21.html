Create a class showing an example of parameterized constructor.


#include <iostream>
using namespace std;

class Student {
public:
    int age;
    Student(int a) {
        age = a;  
    }

    void show() {
        cout << "Age: " << age << endl;
    }
};

int main() {
    Student s1(20);  // passing 20 to constructor
    Student s2(25);  // passing 25 to constructor

    s1.show();
    s2.show();
    return 0;
}

Output


Age: 20
Age: 25
