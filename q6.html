Write a C++ program to create a class called "Employee" with a name, job title, and salary attributes, and methods to calculate and update salary.


#include<iostream>
using namespace std;

class Employee
{
public:
    char name[50], job[50];
    float salary;

    void addEmployee(char n[], char j[], float s)
    {
        // copy values
        strcpy(name, n);
        strcpy(job, j);
        salary = s;
    }

    void updateSalary(float s)
    {
        salary = salary + s;
    }

    void display()
    {
        cout << "Name: " << name << endl;
        cout << "Job: " << job << endl;
        cout << "Salary: " << salary << endl;
    }
};

int main()
{
    Employee e;

    e.addEmployee((char*)"Ravi", (char*)"Manager", 20000);

    cout << "Before update:\n";
    e.display();

    e.updateSalary(5000);

    cout << "\nAfter update:\n";
    e.display();

    return 0;
}

Output


Before update:
Name: Ravi
Job: Manager
Salary: 20000

After update:
Name: Ravi
Job: Manager
Salary: 25000
