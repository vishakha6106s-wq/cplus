Write a C++ program to create a class called "Library" with a collection of books and methods to add and remove books.


#include<iostream>
using namespace std;

class Book
{
public:
    char name[20];
};

class Library
{
public:
    Book b1, b2;   // 2 books
};

int main()
{
    Library l;

    // input books
    cout<<"Enter first book name: ";
    cin>>l.b1.name;

    cout<<"Enter second book name: ";
    cin>>l.b2.name;

    // display before removing
    cout<<"\nBooks:\n";
    cout<<l.b1.name<<endl;
    cout<<l.b2.name<<endl;

    // remove second book
    l.b2.name[0] = '\0';

    // display after removing
    cout<<"\nAfter removing:\n";
    cout<<l.b1.name<<endl;
    cout<<l.b2.name<<endl;

    return 0;
}

Output


Enter first book name: java
Enter second book name: cpp
Books:
java
cpp
After removing:
java
