Program to determine whether one string is a rotation of another


#include<iostream>
using namespace std;

int main()
{
    char s1[50], s2[50], temp[100];
    int i, j, found = 0;

    cout<<"Enter first string: ";
    cin>>s1;

    cout<<"Enter second string: ";
    cin>>s2;

    // copy s1
    for(i=0; s1[i]!='\0'; i++)
    {
        temp[i] = s1[i];
    }

    // add s1 again
    for(j=0; s1[j]!='\0'; j++)
    {
        temp[i+j] = s1[j];
    }

    temp[i+j] = '\0';

    // check s2 inside temp
    for(i=0; temp[i]!='\0'; i++)
    {
        for(j=0; s2[j]!='\0'; j++)
        {
            if(temp[i+j] != s2[j])
                break;
        }

        if(s2[j]=='\0')
        {
            found = 1;
            break;
        }
    }

    if(found==1)
        cout<<"Rotation";
    else
        cout<<"Not Rotation";

    return 0;
}
