//Write a C++ method to print characters between two characters (i.e. A to P).
#include<iostream>
#include<cstring>
using namespace  std;
void printchar(){
    for(char c='A'; c<='P'; c++){
        cout<<c<<"  ";
    }}
int main(){
    printchar();
    return 0;
}
