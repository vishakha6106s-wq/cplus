Write a C++ program to create a class called "Inventory" with a collection of products and methods to add and remove products, and to check for low inventory.


#include<iostream>
using namespace std;

int main()
{
    char name1[20], name2[20];
    int q1, q2;

    // input
    cout<<"Enter product1 name: ";
    cin>>name1;
    cout<<"Enter product1 quantity: ";
    cin>>q1;

    cout<<"Enter product2 name: ";
    cin>>name2;
    cout<<"Enter product2 quantity: ";
    cin>>q2;

    // check low stock
    if(q1 < 5)
        cout<<name1<<" in low stock"<<endl;

    if(q2 < 5)
        cout<<name2<<"in low stock"<<endl;

    // remove second product
    q2 = 0;

    // display
    cout<<"\nProducts:\n";
    cout<<name1<<" "<<q1<<endl;
    cout<<name2<<" "<<q2<<endl;

    return 0;
}
