Write a C++ program to find the index of an array element.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,element,index=-1;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Enter element of find:";
cin>>element;
for(i=0;i<n;i++)
{
if(arr[i]==element)
{
index=i;
break;
}
}
if(index!=-1)
cout<<"Element "<<element<<" found at index:"<<index<<endl;
else
cout<<"Element"<<element<<"not found in the array:"<<index<<endl;
return 0;
}
