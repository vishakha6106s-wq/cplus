Program to Add Characters to a String


#include<iostream>
using namespace std;

int main()
{
    char str[50];
    char ch;
    int i;

    cout<<"Enter string: ";
    cin>>str;

    cout<<"Enter character: ";
    cin>>ch;

    // find end of string
    for(i=0; str[i] != '\0'; i++)
    {
        // just move forward
    }

    // add character
    str[i] = ch;
    i++;
    str[i] = '\0';

    cout<<"New string: "<<str;

    return 0;
}

Output


Enter string: bye
Enter character: h
New string: byeh
