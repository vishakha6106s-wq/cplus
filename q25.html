Write a C++ program to create a class called Person with private instance variables name, age. and country. Provide public getter and setter methods to access and modify these variables.


#include <iostream>
#include <cstring> // For strcpy
using namespace std;

class Person {
private:
    char name[50];
    int age;
    char country[50];

public:
    // Setter methods
    void setName(char n[]) {
        strcpy(name, n);
    }

    void setAge(int a) {
        age = a;
    }

    void setCountry(char c[]) {
        strcpy(country, c);
    }

    // Getter methods
    char* getName() {
        return name;
    }

    int getAge() {
        return age;
    }

    char* getCountry() {
        return country;
    }
};

int main() {
    Person p;
    char n[50], c[50];
    int a;
    cout << "Enter name: ";
    cin >> n;
    p.setName(n);
    cout << "Enter age: ";
    cin >> a;
    p.setAge(a);
    cout << "Enter country: ";
    cin >> c;
    p.setCountry(c);
    cout << "\nPerson Details:\n";
    cout << "Name: " << p.getName() << endl;
    cout << "Age: " << p.getAge() << endl;
    cout << "Country: " << p.getCountry() << endl;
    return 0;
}
