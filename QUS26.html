Write a C++ program to get the difference betweem the largest and smallest values in an array of integers


#include<iostream>
using namespace std;
int main()
{
int n,i,arr[20],maxval,minval;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
minval=arr[0];
maxval=arr[0];
for(i=0;i<n;i++)
{
if(arr[i]<minval)
minval=arr[i];
if(arr[i]>maxval)
maxval=arr[i];
}
int difference =maxval-minval;
cout<<"Largest value:"<<maxval<<endl;
cout<<"Smallest value:"<<minval<<endl;
cout<<"Difference:"<<difference<<endl;
return 0;
}
