Program to print the smallest element in an array.


#include<iostream>
using namespace std;
int main()
{
int a[20],n,i,smallest;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:"<<endl;
for(i=0;i<n;i++)
{
cin>>a[i];
}
smallest=a[0];
for(i=0;i<n;i++)
{
if(a[i]<smallest)
{
smallest=a[i];
}
}
cout<<"Smallest element of an array is:"<<smallest;
return 0;
}
