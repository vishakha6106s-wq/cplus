Write a C++ program to create a class called "Movie" with attributes for title, director, actors, and reviews, and methods for adding and retrieving reviews.


#include <iostream>
#include <string>
using namespace std;

class Movie {
public:
    string title;
    string director;
    string actors[5]; // max 5 actors
    string reviews[5]; // max 5 reviews
    int reviewCount;

    Movie() {
        reviewCount = 0; // start with 0 reviews
    }

    void addReview(string r) {
        if (reviewCount < 5) {
            reviews[reviewCount] = r;
            reviewCount++;
        } else {
            cout << "Review limit reached\n";
        }
    }

    void showReviews() {
        cout << "Reviews for " << title << ":\n";
        for (int i = 0; i < reviewCount; i++) {
            cout << i+1 << ". " << reviews[i] << endl;
        }
    }
};

int main() {
    Movie m;

    // Input movie details
    cout << "Enter movie title: ";
    getline(cin, m.title);
    cout << "Enter director: ";
    getline(cin, m.director);

    cout << "Enter 3 actors:\n";
    for(int i=0; i<3; i++) {
        cout << "Actor " << i+1 << ": ";
        getline(cin, m.actors[i]);
    }

    // Add reviews
    m.addReview("Amazing movie!");
    m.addReview("Good acting and story.");

    // Display reviews
    cout << "\nMovie: " << m.title << endl;
    cout << "Director: " << m.director << endl;
    cout << "Actors: ";
    for(int i=0; i<3; i++) cout << m.actors[i] << " ";
    cout << "\n";
    m.showReviews();

    return 0;
}
