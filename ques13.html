Program to find all the permutations of a string


#include<iostream>
using namespace std;

void swap(char &a, char &b)
{
    char temp = a;
    a = b;
    b = temp;
}

void permute(char str[], int l, int r)
{
    int i;

    if(l == r)
    {
        cout<<str<<endl;
        return;
    }

    for(i = l; i <= r; i++)
    {
        swap(str[l], str[i]);
        permute(str, l+1, r);
        swap(str[l], str[i]);   // back
    }
}

int main()
{
    char str[10];

    cout<<"Enter string: ";
    cin>>str;

    int len = 0;
    while(str[len] != '\0')
        len++;

    permute(str, 0, len-1);

    return 0;
}
