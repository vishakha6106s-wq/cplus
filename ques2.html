Program to Check if a String contains only digits?


#include<iostream>
using namespace std;

int main()
{
    char str[50];
    int i, flag = 1;

    cout<<"Enter string: ";
    cin>>str;

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i]<'0' || str[i]>'9')
        {
            flag = 0;
            break;
        }
    }

    if(flag == 1)
        cout<<"Only digits";
    else
        cout<<"Not only digits";

    return 0;
}

Output


Enter string: 234
Only digits
