Program to find the largest and smallest word in a string


#include<iostream>
using namespace std;
int main()
{
    char str[100];
    int i=0, len=0;
    int max=0, min=100;

    cout<<"Enter string: ";
    cin.getline(str,100);

    // find max and min length
    while(str[i]!='\0')
    {
        if(str[i]==' ')
        {
            if(len > max) max = len;
            if(len < min) min = len;
            len = 0;
        }
        else
        {
            len++;
        }
        i++;
    }

    if(len > max) max = len;
    if(len < min) min = len;

    // print largest word
    cout<<"Largest word: ";
    i=0; len=0;
    while(str[i]!='\0')
    {
        if(str[i]==' ')
        {
            if(len == max)
            {
                for(int j=i-len; j<i; j++)
                    cout<<str[j];
                break;
            }
            len=0;
        }
        else len++;

        i++;
    }

    // print smallest word
    cout<<endl<<"Smallest word: ";
    i=0; len=0;
    while(str[i]!='\0')
    {
        if(str[i]==' ')
        {
            if(len == min)
            {
                for(int j=i-len; j<i; j++)
                    cout<<str[j];
                break;
            }
            len=0;
        }
        else len++;

        i++;
    }

    return 0;
}
