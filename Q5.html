Define a class Date to represent dates. Overload the < operator to compare two dates



#include <iostream>
using namespace std;

class Date {
    int d, m, y;

public:
    Date(int dd=0, int mm=0, int yy=0) {
        d = dd; m = mm; y = yy;
    }

    bool operator<(Date x) {
        if(y < x.y) return true;
        if(y == x.y && m < x.m) return true;
        if(y == x.y && m == x.m && d < x.d) return true;
        return false;
    }
};

int main() {
    Date a(1,1,2020), b(2,1,2020);
    if(a < b)
        cout << "a is earlier";
    else
        cout << "b is earlier";
}

Output


a is earlier
