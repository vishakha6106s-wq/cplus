Write a C++ program to find the common elements between two arrays.


#include<iostream>
using namespace std;
int main()
{
int arr1[20],arr2[20],arr,n1,n2,i,j;
cout<<"Enter the first size of an array:";
cin>>n1;
cout<<"Enter the elements:";
for(i=0;i<n1;i++)
{
cin>>arr1[i];
}
cout<<"Enter the second size of an array:";
cin>>n2;
cout<<"Enter the elements:";
for(i=0;i<n2;i++)
{
cin>>arr2[i];
}
cout<<"Common elements are:";
for(i=0;i<n1;i++)
{
for(j=0;j<n2;j++)
{
if(arr1[i]==arr2[j])
{
cout<<arr1[i]<< " ";
break;
}
}
}
return 0;
}
