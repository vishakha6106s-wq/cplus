Program to Swapping Pair of Characters


#include<iostream>
using namespace std;

int main()
{
    char str[50];
    char temp;

    cout<<"Enter string: ";
    cin>>str;

    for(int i=0; str[i]!='\0'; i=i+2)
    {
        if(str[i+1]!='\0')
        {
            temp = str[i];
            str[i] = str[i+1];
            str[i+1] = temp;
        }
    }

    cout<<"After swapping: "<<str;

    return 0;
}

Output


Enter string: hello
After swapping: ehllo
