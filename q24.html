Create a class showing the area of circle and rectangle by method overloading.


#include <iostream>
using namespace std;

class Shape {
public:
    void area(float radius) {
        cout << "Area of Circle: " << 3.14 * radius * radius << endl;
    }

    void area(float length, float width) {
        cout << "Area of Rectangle: " << length * width << endl;
    }
};

int main() {
    Shape s;
    float r, l, w;
    cout << "Enter radius of circle: ";
    cin >> r;
    s.area(r);

    cout << "Enter length of rectangle: ";
    cin >> l ;
    cout<<"Enter width: ";
    cin>> w;
    s.area(l, w);
    return 0;
}

Output


Enter radius of circle: 34.55
Area of Circle: 3748.23
Enter length of rectangle: 23
Enter width: 46.7
Area of Rectangle: 1074.1
