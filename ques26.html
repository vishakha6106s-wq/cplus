Program to replace lower-case characters with upper-case and vice-versa


#include<iostream>
using namespace std;
int main()
{
    char str[100];
    int i;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i] >= 'a' && str[i] <= 'z')
            str[i] = str[i] - 32;   // small → capital

        else if(str[i] >= 'A' && str[i] <= 'Z')
            str[i] = str[i] + 32;   // capital → small
    }

    cout<<"New string: "<<str;

    return 0;
}

Output


Enter string: hello
New string: HELLO
