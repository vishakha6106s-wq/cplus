Write a C++ program to insert an element (specific position) into an array.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,pos,element;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Enter element to insert:";
cin>>element;
cout<<"Enter position to insert (0-based index):";
cin>>pos;
if(pos<0||pos>n)
{
cout<<"Invalid position!"<<endl;
}
for(i=n;i>pos;i--)
{
arr[i]=arr[i-1];
}
arr[pos]=element;
n++;
cout<<"Array after insertion:";
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
