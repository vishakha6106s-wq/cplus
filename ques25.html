Program to remove all the white spaces from a string


#include<iostream>
using namespace std;
int main()
{
    char str[100];
    int i;

    cout<<"Enter string: ";
    cin.getline(str,100);

    cout<<"New string: ";

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i] != ' ')
            cout<<str[i];
    }

    return 0;
}

Output


Enter string: hello bye
New string: hellobye
