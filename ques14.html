Program to find maximum and minimum occurring character in a string


#include<iostream>
using namespace std;

int main()
{
    char str[50];
    int i, j, count, max=0, min=100;
    char maxc, minc;

    cout<<"Enter string: ";
    cin>>str;

    for(i=0; str[i]!='\0'; i++)
    {
        count = 0;

        for(j=0; str[j]!='\0'; j++)
        {
            if(str[i] == str[j])
                count++;
        }

        if(count > max)
        {
            max = count;
            maxc = str[i];
        }

        if(count < min)
        {
            min = count;
            minc = str[i];
        }
    }

    cout<<"Max char: "<<maxc<<endl;
    cout<<"Min char: "<<minc;

    return 0;
}

Enter string: hello
Max char: l
Min char: h
