Write a C++ method to count all vowels in a string


//Write a C++ method to count all vowels in a string
#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i, count = 0;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i]=='a' || str[i]=='e' || str[i]=='i' || 
           str[i]=='o' || str[i]=='u' ||
           str[i]=='A' || str[i]=='E' || str[i]=='I' || 
           str[i]=='O' || str[i]=='U')
        {
            count++;
        }
    }

    cout<<"Total vowels: "<<count;

    return 0;
}/*OUTPUT
Enter string: hello
Total vowels: 2
*/
