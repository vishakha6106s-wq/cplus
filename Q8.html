Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.



#include <iostream>
using namespace std;

class Time {
    int h, m;

public:
    Time(int hh=0, int mm=0) {
        h = hh; m = mm;
    }

    friend ostream& operator<<(ostream &out, Time t) {
        out << t.h << ":" << t.m;
        return out;
    }
};

int main() {
    Time t(10, 30);
    cout << t;
}

Output


10:30
