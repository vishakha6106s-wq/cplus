Implement a class String to represent strings. Overload the + operator to concatenate two strings.



#include <iostream>
#include <cstring>
using namespace std;

class String {
    char s[50];

public:
    String(char str[] = "") {
        strcpy(s, str);
    }

    String operator+(String t) {
        String r;
        strcpy(r.s, s);
        strcat(r.s, t.s);
        return r;
    }

    void show() {
        cout << s;
    }
};

int main() {
    String a("Hello "), b("World"), c;
    c = a + b;
    c.show();
}

Output


Hello World
