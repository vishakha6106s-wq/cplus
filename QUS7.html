Program to print the elements of an array present on even position.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Even Position"<<endl;
for(i=0;i<n;i++)
{
if(arr[i]%2==0)
{
cout<<arr[i]<<endl;
}
}
return 0;
}
