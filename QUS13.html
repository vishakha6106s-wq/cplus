Program to sort the elements of an array in ascending order.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,j,temp;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
for(i=0;i<n-1;i++)
{
for(j=i+1;j<n;j++)
{
if(arr[i]>arr[j]);
{
temp=arr[i];
arr[i]=arr[j];
arr[j]=temp;
}
}
}
cout<<"Array in ascending order:";
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
