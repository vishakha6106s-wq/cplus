Program to Print even length words


#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i=0, len=0, start=0;

    cout<<"Enter string: ";
    cin.getline(str,100);

    while(str[i] != '\0')
    {
        if(str[i] == ' ')
        {
            if(len % 2 == 0)
            {
                for(int j=start; j<i; j++)
                    cout<<str[j];
                cout<<endl;
            }
            len = 0;
            start = i + 1;
        }
        else
        {
            len++;
        }
        i++;
    }

    // last word
    if(len % 2 == 0)
    {
        for(int j=start; j<i; j++)
            cout<<str[j];
    }

    return 0;
}
