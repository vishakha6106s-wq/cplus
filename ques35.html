Write a C++ method to count all words in a string


//Write a C++ method to count all words in a string
#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i, count = 1;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i]==' ')
            count++;
    }

    cout<<"Total words: "<<count;

    return 0;
}/*OUTPUT
Enter string: I am learning cpp
Total words: 4
*/
