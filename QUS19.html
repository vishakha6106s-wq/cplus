Write a C++ program to swap the first and last elements of an array and create a new array.


#include<iostream>
using namespace std;
int main()
{
int arr[20],newArr[50],n,i;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
for(i=0;i<n;i++)
{
newArr[i]=arr[i];
}
if(n>1)
{
int temp=newArr[0];
newArr[0]=newArr[n-1];
newArr[n-1]=temp;
}
cout<<"New Array after swapping first and last elements:";
for(i=0;i<n;i++)
{
cout<<newArr[i]<<" ";
}
return 0;
}
