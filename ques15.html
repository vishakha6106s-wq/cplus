Program to find Reverse of the string


#include<iostream>
#include<cstring>
using namespace std;
int main()
{
   
    char arr[50],str[50];
    int i,l,j=0;
    cout<<"Enter a string : ";
    cin.getline(arr,50);
    
    l=strlen(arr);
    for(i=l-1 ; i>=0 ; i-- )
    {
        str[j]=arr[i];
        j++;
    }
    str[j] = '\0'; 
    cout<<str;
    return 0;
}
