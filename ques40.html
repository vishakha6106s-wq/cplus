Write a C++ method to check whether a string is a valid password.


//Write a C++ method to check whether a string is a valid password.
#include<iostream>
using namespace std;

int checkPassword(char str[])
{
    int i=0, len=0, digit=0;

    while(str[i] != '\0')
    {
        len++;

        if(str[i] >= '0' && str[i] <= '9')
            digit = 1;

        i++;
    }

    if(len >= 6 && digit == 1)
        return 1;
    else
        return 0;
}

int main()
{
    char pass[50];

    cout<<"Enter password: ";
    cin>>pass;

    if(checkPassword(pass))
        cout<<"Valid Password";
    else
        cout<<"Invalid Password";

    return 0;
}/*output
Enter password: abc123
Valid Password
