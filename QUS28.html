Write a C++ program to cyclically rotate a given array clockwise by one.


#include<iostream>
using namespace std;
int main()
{
int n,i,arr[20],last;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
last=arr[n-1];
for(i=n-1;i>0;i--)
{
arr[i]=arr[i-1];
}
arr[0]=last;
cout<<"Array after cyclic rotation:"<<endl;
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
