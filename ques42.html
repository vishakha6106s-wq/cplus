//Write a C++ method that accepts three integers and returns true if one is the middle point between the other two integers, otherwise false
#include<iostream>
using namespace std;

bool check(int a, int b, int c)
{
    if((a>b && a<c) || (a<c && a>b))
        return true;

    if((b>a && b<c) || (b<c && b>a))
        return true;

    if((c>a && c<b) || (c<b && c>a))
        return true;

    return false;
}

int main()
{
    int a,b,c;

    cout<<"Enter 3 numbers: ";
    cin>>a>>b>>c;

    if(check(a,b,c))
        cout<<"True";
    else
        cout<<"False";

    return 0;
}/*OUTPUT
Enter 3 numbers: 2 5 8
True
Enter 3 numbers: 5 5 5
False
*/
