Write a C++ program to create a class called "TrafficLight" with attributes for color and duration, and methods to change the color and check for red or green.


 #include<iostream>
using namespace std;

class TrafficLight
{
public:
    char color[10];
    int duration;
};

int main()
{
    TrafficLight t;

    // input
    cout<<"Enter color and duration: ";
    cin>>t.color>>t.duration;

    // change color
    cout<<"Enter new color: ";
    cin>>t.color;

    // check color
    if(t.color[0]=='r')
        cout<<"Red light";
    else if(t.color[0]=='g')
        cout<<"Green light";
    else
        cout<<"Other color";

    return 0;
}

Output


Enter color and duration: blue 20
Enter new color: red
Red light
