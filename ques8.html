Program to count the total number of characters in a string


#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i, count = 0;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i] != '\0'; i++)
    {
        count++;
    }

    cout<<"Total characters: "<<count;

    return 0;
}

Output


Enter string: Hello
Total characters: 5
