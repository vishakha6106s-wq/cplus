Program to left rotate the element of an array.


#include<iostream>
using namespace std;
int main()
{
int arr[50],n,i,temp;
cout<<"Enter number of elements:";
cin>>n;
cout<<"Enter elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
temp=arr[0];
for(i=0;i<n-1;i++)
{
arr[i]= arr[i+1];
}
arr[n-1]=temp;
cout<<"Array after left rotation:";
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
