Write a C++ program to test if an array contains a specific value.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,value;
int found=0;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Enter value to search:";
cin>>value;
for(i=0;i<n;i++)
{
if(arr[i]==value)
{
found=1;
break;
}
}
if(found)
cout<<"Array contains the value:"<<value<<endl;
else
cout<<"Array does not contains the value"<<value<<endl;
return 0;
}
