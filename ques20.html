Program to Get a Character From the Given String


#include<iostream>
using namespace std;
int main()
{
    char str[50];
    int i;

    cout<<"Enter string: ";
    cin>>str;

    cout<<"Characters are: ";

    for(i=0; str[i]!='\0'; i++)
    {
        cout<<str[i]<<"  ";
    }

    return 0;
}

Output


Enter string: hello
Characters are: h  e  l  l  o
