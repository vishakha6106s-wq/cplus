Program to remove all occurrences of a given character from input String?


#include<iostream>
using namespace std;

int main()
{
    char str[50], ch;
    int i, j=0;

    cout<<"Enter string: ";
    cin>>str;

    cout<<"Enter character to remove: ";
    cin>>ch;

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i] != ch)
        {
            str[j] = str[i];
            j++;
        }
    }

    str[j] = '\0';

    cout<<"After removing: "<<str;

    return 0;
}

Output


Enter string: hello
Enter character to remove: o
After removing: hell
