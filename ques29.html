Program to Splitting into a number of sub-strings


//Program to Splitting into a number of sub-strings
#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int n, len=0, i, count=0;

    cout<<"Enter string: ";
    cin.getline(str,100);

    cout<<"Enter number of parts: ";
    cin>>n;

    // find length
    for(i=0; str[i]!='\0'; i++)
        len++;

    if(len % n != 0)
    {
        cout<<"Not possible";
        return 0;
    }

    int size = len / n;

    cout<<"Substrings:"<<endl;

    for(i=0; str[i]!='\0'; i++)
    {
        cout<<str[i];
        count++;

        if(count == size)
        {
            cout<<endl;
            count = 0;
        }
    }

    return 0;
}
/*OUTPUT
Enter string: abcdef
Enter number of parts: 3
Substrings:
ab
