Write a C++ program to create an abstract class Animal with an abstract method called sound(). Create derived classes Lion and Tiger that extend the Animal class and implement the sound() method to make a specific sound for each animal.


#include <iostream>
#include <string.h>
using namespace std;

class animal
{
protected:
    virtual void sound() = 0;
};
class tiger : public animal
{
public:
    void sound()
    {
        cout << "Tiger roars." << endl;
    }
};
class lion : public animal
{
public:
    void sound()
    {
        cout << "Lion roars loudly." << endl;
    }
};

int main()
{
    tiger t1;
    lion l1;

    t1.sound();
    cout << endl;
    l1.sound();
    return 0;
}

Output


Tiger roars
Lion roars loudly
