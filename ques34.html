Write a program to find out first non repeated character from input String?


//Write a program to find out first non repeated character from input String?
#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i, j, count;

    cout<<"Enter string: ";
    cin>>str;

    for(i=0; str[i]!='\0'; i++)
    {
        count = 0;

        for(j=0; str[j]!='\0'; j++)
        {
            if(str[i] == str[j])
                count++;
        }

        if(count == 1)
        {
            cout<<"First non-repeated character: "<<str[i];
            return 0;
        }
    }

    cout<<"No non-repeated character";

    return 0;
}/*OUTPUT
Enter string: hello
First non-repeated character: h

Enter string: aabbcc
No non-repeated character
*/
