//Write a C++ method to display the current date and time.
#include<iostream>
#include<ctime>
using namespace std;

void show()
{
    time_t t = time(0);
    tm *x = localtime(&t);

    cout<<"Date: "<<x->tm_mday<<"/"<<x->tm_mon+1<<"/"<<x->tm_year+1900<<endl;
    cout<<"Time: "<<x->tm_hour<<":"<<x->tm_min<<":"<<x->tm_sec;
}

int main()
{
    show();
    return 0;
}
/*OUTPUTDate: 28/3/2026
Time: 23:45:30% 
*/
