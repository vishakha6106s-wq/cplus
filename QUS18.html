Program to Remove Duplicate Element in an array.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,j,k;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
for(i=0;i<n;i++)
{
for(j=i+1;j<n;j++)
{
if(arr[i]==arr[j])
{
for(k=j;k<n-1;k++)
{
arr[k]=arr[k+1];
}
n--;
}
else
{
j++;
}
}
}
cout<<"Array after removing duplicates:";
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
