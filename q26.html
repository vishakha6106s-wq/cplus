Write a C++ program to create a class called BankAccount with private instance variables accountNumber and balance. Provide public getter and setter methods to access and modify these variables.


#include <iostream>
#include <cstring>
using namespace std;

class BankAccount {
private:
    char accountNumber[20]; // Account number as char array
    int balance;            // Balance as integer

public:
    // Setter methods
    void setAccountNumber(char acc[]) {
        strcpy(accountNumber, acc);
    }

    void setBalance(int b) {
        balance = b;
    }

    // Getter methods
    char* getAccountNumber() {
        return accountNumber;
    }

    int getBalance() {
        return balance;
    }
};

int main() {
    BankAccount b;
    char acc[20];
    int bal;

    cout << "Enter account number: ";
    cin >> acc;
    b.setAccountNumber(acc);

    cout << "Enter balance: ";
    cin >> bal;
    b.setBalance(bal);

    cout << "\nBank Account Details:\n";
    cout << "Account Number: " << b.getAccountNumber() << endl;
    cout << "Balance: " << b.getBalance() << endl;

    return 0;
}

Output


Enter account number: 12345
Enter balance: 5000
Bank Account Details:
Account Number: 12345
Balance: 5000
