Write a C++ program to create a class called Car with private instance variables company_name, model_name, year, and mileage. Provide public getter and setter methods to access and modify the company_name, model_name, and year variables. However, only provide a getter method for the mileage variable.


#include <iostream>
#include <cstring>
using namespace std;

class Car
{
private:
    char company[50];
    char model[50];
    int year;
    int mileage;

public:
    // Setter methods
    void setCompany(char c[])
    {
        strcpy(company, c);
    }
    void setModel(char m[]) 
    {
         strcpy(model, m); 
    }
    void setYear(int y)
    {
         year = y; 
    }
    void setMileage(int m) { 
        mileage = m; 
    } 

    // Getter methods
    char *getCompany() { 
        return company; 
    }
    char *getModel() { 
        return model;
     }
    int getYear() {
         return year;
         }
    int getMileage() {
         return mileage; 
        } 
};

int main()
{
    Car c1;
    char str[50];
    int y, m;

    cout << "Enter Company Name: ";
    cin >> str;
    c1.setCompany(str);

    cout << "Enter Model Name: ";
    cin >> str;
    c1.setModel(str);

    cout << "Enter Year: ";
    cin >> y;
    c1.setYear(y);

    cout << "Enter Mileage: ";
    cin >> m;
    c1.setMileage(m);

    cout << "\nCar Details:\n";
    cout << "Company: " << c1.getCompany() << endl;
    cout << "Model: " << c1.getModel() << endl;
    cout << "Year: " << c1.getYear() << endl;
    cout << "Mileage: " << c1.getMileage() << " km/l" << endl;

    return 0;
}
