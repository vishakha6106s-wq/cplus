Write a C++ program to create a class called "Book" with attributes for title, author, and ISBN, and methods to add and remove books from a collection.


#include<iostream>
using namespace std;
class Book
{
public:
    string title, author, isbn;

    void addBook(string t, string a, string i)
    {
        title = t;
        author = a;
        isbn = i;
    }

    void removeBook()
    {
        title = "";
        author = "";
        isbn = "";
    }

    void display()
    {
        cout << "Title: " << title << endl;
        cout << "Author: " << author << endl;
        cout << "ISBN: " << isbn << endl;
    }
};

int main()
{
    Book b;

    b.addBook("C++ Basics", "Ravi", "12345");  // add book
    b.display();

    b.removeBook();  // remove book

    cout << "\nAfter removing:\n";
    b.display();

    return 0;
}

ISBN: 12345
