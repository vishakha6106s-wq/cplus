Program to separate the Individual Characters from a String


#include<iostream>
using namespace std;
int main()
{
    char str[100];
    int i;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i]!='\0'; i++)
    {
        cout<<str[i]<<endl;
    }

    return 0;
}
