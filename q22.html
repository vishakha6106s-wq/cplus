Create a class showing an example of copy constructor.


#include <iostream>
using namespace std;

class Student {
private:
    int age;

public:
    // Constructor
    Student(int a) {
        age = a;
    }
    // Copy Constructor
    Student(const Student &s) {
        age = s.age;
    }
    void display() {
        cout << "Age: " << age << endl;
    }
};

int main() {
    Student s1(20);     // Original object
    Student s2 = s1;   // Copy constructor called

    cout << "Original Student: ";
    s1.display();

    cout << "Copied Student: ";
    s2.display();

    return 0;
}

Output


Original Student: Age: 20
Copied Student: Age: 20
