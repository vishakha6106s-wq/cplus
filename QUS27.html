Write a C++ program to segregate all 0s on left side and all 1s on right side of a given array of 0s and 1s.


#include<iostream>
using namespace std;
int main()
{
int n,i,arr[20],left,right;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter "<<n<<" elements (0s and 1s only):";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
left=0,right=n-1;
while(left<right)
{
while(arr[left]==0&&left<right)
left++;
while(arr[right]==1&&left<right)
right--;
if(left<right)
{
int temp=arr[left];
arr[left]=arr[right];
arr[right]=temp;
left++;
right--;
}
}
cout<<"Array after segregation"<<endl;
for(i=0;i<n;i++)
{
cout<<arr[i]<<" ";
}
return 0;
}
