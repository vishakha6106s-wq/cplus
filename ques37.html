Write a C++ Program to Sort Names in an Alphabetical Order


//Write a C++ Program to Sort Names in an Alphabetical Order
#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    char name[5][20];
    int i, j;

    cout<<"Enter 5 names:\n";
    for(i=0; i<5; i++)
        cin>>name[i];

    // sorting
    for(i=0; i<5; i++)
    {
        for(j=i+1; j<5; j++)
        {
            if(strcmp(name[i], name[j]) > 0)
            {
                char temp[20];
                strcpy(temp, name[i]);
                strcpy(name[i], name[j]);
                strcpy(name[j], temp);
            }
        }
    }

    cout<<"\nSorted names:\n";
    for(i=0; i<5; i++)
        cout<<name[i]<<endl;

    return 0;
}/*OUTPUT
Enter 5 names:
Ravi
Aman
Simran
Karan
Neha

Sorted names:
Aman
Karan
Neha
Ravi
Simran */
