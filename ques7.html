Program to count number of words in a String?


#include<iostream>
using namespace std;

int main()
{
    char str[100];
    int i, count = 1;

    cout<<"Enter string: ";
    cin.getline(str,100);

    for(i=0; str[i] != '\0'; i++)
    {
        if(str[i] == ' ')
            count++;
    }

    cout<<"Number of words: "<<count;

    return 0;
}

Output


Enter string: I am student
Number of words: 3
