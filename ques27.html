Program to replace the spaces of a string with a specific character


#include<iostream>
using namespace std;
int main()
{
    char str[100], ch;
    int i;

    cout<<"Enter string: ";
    cin.getline(str,100);

    cout<<"Enter character: ";
    cin>>ch;

    for(i=0; str[i]!='\0'; i++)
    {
        if(str[i]==' ')
            str[i]=ch;
    }

    cout<<"New string: "<<str;
    return 0;
}

Output


Enter string: hello bye
Enter character: _
New string: hello_bye
