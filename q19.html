Write a C++ program to create a class with methods to search for flights and hotels, and to book and cancel reservations.


#include <iostream>
#include <string>
using namespace std;

class Reservation {
public:
    string flight[5];
    string hotel[5];
    bool flightBooked[5];
    bool hotelBooked[5];
    int flightCount;
    int hotelCount;

    Reservation() {
        flightCount = 0;
        hotelCount = 0;
        for(int i=0;i<5;i++) { flightBooked[i]=false; hotelBooked[i]=false; }
    }

    void addFlight(string name) {
        if(flightCount<5) flight[flightCount++] = name;
    }

    void addHotel(string name) {
        if(hotelCount<5) hotel[hotelCount++] = name;
    }

    void searchFlights(string name) {
        bool found=false;
        for(int i=0;i<flightCount;i++) {
            if(flight[i]==name) { cout<<"Flight found: "<<name<<endl; found=true; break; }
        }
        if(!found) cout<<"Flight not found\n";
    }

    void searchHotels(string name) {
        bool found=false;
        for(int i=0;i<hotelCount;i++) {
            if(hotel[i]==name) { cout<<"Hotel found: "<<name<<endl; found=true; break; }
        }
        if(!found) cout<<"Hotel not found\n";
    }

    void bookFlight(string name) {
        for(int i=0;i<flightCount;i++) {
            if(flight[i]==name && !flightBooked[i]) {
                flightBooked[i]=true;
                cout<<"Flight booked: "<<name<<endl;
                return;
            }
        }
        cout<<"Flight not available or already booked\n";
    }

    void bookHotel(string name) {
        for(int i=0;i<hotelCount;i++) {
            if(hotel[i]==name && !hotelBooked[i]) {
                hotelBooked[i]=true;
                cout<<"Hotel booked: "<<name<<endl;
                return;
            }
        }
        cout<<"Hotel not available or already booked\n";
    }

    void cancelFlight(string name) {
        for(int i=0;i<flightCount;i++) {
            if(flight[i]==name && flightBooked[i]) {
                flightBooked[i]=false;
                cout<<"Flight cancelled: "<<name<<endl;
                return;
            }
        }
        cout<<"Flight not booked or not found\n";
    }

    void cancelHotel(string name) {
        for(int i=0;i<hotelCount;i++) {
            if(hotel[i]==name && hotelBooked[i]) {
                hotelBooked[i]=false;
                cout<<"Hotel cancelled: "<<name<<endl;
                return;
            }
        }
        cout<<"Hotel not booked or not found\n";
    }
};

int main() {
    Reservation r;

    // Add some flights and hotels
    r.addFlight("Flight101");
    r.addFlight("Flight202");
    r.addHotel("HotelA");
    r.addHotel("HotelB");

    // Test
    r.searchFlights("Flight101");
    r.bookFlight("Flight101");
    r.bookFlight("Flight101"); // already booked
    r.cancelFlight("Flight101");

    r.searchHotels("HotelA");
    r.bookHotel("HotelA");
    r.cancelHotel("HotelA");

    return 0;
}
