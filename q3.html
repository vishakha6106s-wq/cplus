Write a C++ program to create an abstract class BankAccount with abstract methods deposit() and withdraw(). Create derived classes: SavingsAccount and CurrentAccount that extend the BankAccount class and implement the respective methods to handle deposits and withdrawals for each account type.


#include <iostream>
using namespace std;

class bankAccount
{
protected:
    int balance;
    virtual void deposit() = 0;
    virtual void withdraw() = 0;
};

class savingsAccount : public bankAccount
{
public:
    void deposit()
    {
        int amt;
        cout << "Enter the amount you would like to withdraw: ";
        cin >> amt;
        bankAccount::balance += amt;
        cout << "Amount Deposited!" << endl;
        cout << "New Balance: " << bankAccount::balance;
    }
    void withdraw()
    {
        int amt;
        cout << "Enter the amount you would like to withdraw: ";
        cin >> amt;
        if (amt >= (bankAccount::balance - 500))
        {
            bankAccount::balance -= amt;
            cout << "Collect your Cash" << endl;
            cout << "New Balance: " << bankAccount::balance;
        }
        else
        {
            cout << "Insufficient Balance" << endl;
            cout << "Minimum balance in account should be 500";
        }
    }
};

{

        {

        }
        else
        {
            overdraft = bankAccount::balance;
            cout << "Overdraft generated!" << endl;
            cout << "Overdraft Amount: " << overdraft << " (Overdrafft Charges may apply, please deposit the overdraft amount to avoid charges)";
        }
    }
};
int main()
{
    savingsAccount s;
    s.deposit()<<endl;

    return 0;
}
