Write a C++ program to create a class called "School" with attributes for students, teachers, and classes, and methods to add and remove students and teachers, and to create classes.


#include<iostream>
using namespace std;

class School
{
public:
    char student[20];
    char teacher[20];
    char classname[20];
};

int main()
{
    School s;

    // add (input)
    cout<<"Enter student name: ";
    cin>>s.student;

    cout<<"Enter teacher name: ";
    cin>>s.teacher;

    cout<<"Enter class name: ";
    cin>>s.classname;

    // display
    cout<<"\nSchool Data:\n";
    cout<<s.student<<endl;
    cout<<s.teacher<<endl;
    cout<<s.classname<<endl;

    // remove student and teacher
    s.student[0] = '\0';
    s.teacher[0] = '\0';

    cout<<"After removing:\n";
    cout<<s.student<<endl;
    cout<<s.teacher<<endl;
    cout<<s.classname<<endl;

    return 0;
}
