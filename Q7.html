Create a class Rectangle to represent rectangles. Overload the == operator to check if two rectangles are equal.



#include <iostream>
using namespace std;

class Rectangle {
    int l, b;

public:
    Rectangle(int x=0, int y=0) {
        l = x; b = y;
    }

    bool operator==(Rectangle r) {
        return (l == r.l && b == r.b);
    }
};

int main() {
    Rectangle a(4,5), b(4,5);

    if(a == b)
        cout << "Equal";
    else
        cout << "Not Equal";
}

Equal
