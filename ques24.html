Program to print smallest and biggest possible palindrome word in a given string


#include<iostream>
using namespace std;
int main()
{
    char str[100], word[20];
    int i=0, j=0;

    char small[20]="", big[20]="";

    cout<<"Enter string: ";
    cin.getline(str,100);

    while(1)
    {
        if(str[i]!=' ' && str[i]!='\0')
        {
            word[j]=str[i];
            j++;
        }
        else
        {
            word[j]='\0';

            // check palindrome
            int len=j, flag=1;
            for(int k=0; k<len/2; k++)
            {
                if(word[k]!=word[len-1-k])
                    flag=0;
            }

            if(flag==1 && len>0)
            {
                if(small[0]=='\0' || len < strlen(small))
                    strcpy(small,word);

                if(big[0]=='\0' || len > strlen(big))
                    strcpy(big,word);
            }

            j=0;

            if(str[i]=='\0')
                break;
        }
        i++;
    }

    if(small[0]=='\0')
        cout<<"No palindrome word";
    else
    {
        cout<<"Smallest: "<<small<<endl;
        cout<<"Biggest: "<<big;
    }

    return 0;
}
