Program to swap two string variables without using third or temp variable.


//Program to swap two string variables without using third or temp variable.
#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    char a[100], b[100], temp[100];

    cout<<"Enter first string: ";
    cin>>a;

    cout<<"Enter second string: ";
    cin>>b;

    strcpy(temp, a);
    strcpy(a, b);
    strcpy(b, temp);

    cout<<"After swap:"<<endl;
    cout<<"a = "<<a<<endl;
    cout<<"b = "<<b;
    
    return 0;
}/*Enter first string: hey
