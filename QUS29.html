Write a C++ program to print all unique element in an array.


#include<iostream>
using namespace std;
int main()
{
int n,i,arr[20],j,count;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
cout<<"Unique elements in the array are:"<<endl;
for(i=0;i<n;i++)
{
count=0;
}
for(j=0;j<n;j++)
{
if(arr[i]==arr[j])
{
count++;
}
}
if(count==1)
{
cout<<arr[i]<<" ";
}
return 0;
}
