Create a class showing an example of default constructor.


#include <iostream>
using namespace std;

class Student {
public:
    int age;

    // Default constructor
    Student() {
        age = 18; // default value
    }
};

int main() {
    Student s;  // default constructor called automatically
    cout<<"Age: "<<s.age;
    return 0;
}

Output


Age: 18
