Write a C++ method to compute the future investment value at a given interest rate for a specified number of years.


//Write a C++ method to compute the future investment value at a given interest rate for a specified number of years.
#include<iostream>
#include<cmath>
using namespace std;

double futureValue(double p, double r, int n)
{
    return p * pow(1 + r/100, n);
}

int main()
{
    double p, r;
    int n;

    cout<<"Enter amount: ";
    cin>>p;

    cout<<"Enter interest rate: ";
    cin>>r;

    cout<<"Enter years: ";
    cin>>n;

    cout<<"Future value: "<<futureValue(p, r, n);

    return 0;
}/*OUTPUT
Enter amount: 1000
Enter rate: 10
Enter years: 2

Future value: 1210
*/
