Write a C++ program to calculate the average value of array elements.


#include<iostream>
using namespace std;
int main()
{
int arr[20],n,i,sum;
float average;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
sum=arr[0]+arr[1]+arr[2]+arr[3]+arr[4];
average=sum/n;
cout<<"Sum="<<sum<<endl;
cout<<"Average of array elements:"<<average<<endl;
return 0;
}
