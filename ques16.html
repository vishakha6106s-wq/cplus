Program to find the duplicate characters in a string


#include<iostream>
using namespace std;
int main()
{
    char str[50];
    int i, j;

    cout<<"Enter string: ";
    cin>>str;

    cout<<"Duplicate characters: ";

    for(i=0; str[i]!='\0'; i++)
    {
        for(j=i+1; str[j]!='\0'; j++)
        {
            if(str[i] == str[j])
            {
                cout<<str[i]<<" ";
                break;   // print once
            }
        }
    }

    return 0;
}

Enter string: hello
Duplicate characters: l
