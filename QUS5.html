Program to print the elements of an array.


#include<iostream>
using namespace std;
int main()
{
int a[20],n,i;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:"<<endl;
for(i=0;i<n;i++)
{
cin>>a[i];
}
cout<<"Print the elements"<<endl;
for(i=0;i<n;i++)
{
cout<<a[i]<<endl;
}
return 0;
}
