Program to find the frequency of characters


#include<iostream>
using namespace std;
int main()
{
    char str[50];
    int i, j, count;

    cout<<"Enter string: ";
    cin>>str;

    for(i=0; str[i]!='\0'; i++)
    {
        count = 0;

        for(j=0; str[j]!='\0'; j++)
        {
            if(str[i] == str[j])
                count++;
        }

        cout<<str[i]<<" = "<<count<<endl;
    }

    return 0;
}
