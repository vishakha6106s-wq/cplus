Program to print the largest element in an array.


#include<iostream>
using namespace std;
int main()
{
int a[20],n,i,largest;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:"<<endl;
for(i=0;i<n;i++)
{
cin>>a[i];
}
largest=a[0];
for(i=0;i<n;i++)
{
if(a[i]>largest)
{
largest=a[i];
}
}
cout<<"Largest element of an array is:"<<largest;
return 0;
}
