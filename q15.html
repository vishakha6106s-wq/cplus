Write a C++ program to create a class called "MusicLibrary" with a collection of songs and methods to add and remove songs, and to play a random song.


#include<iostream>
using namespace std;

class MusicLibrary
{
public:
    char song1[20];
    char song2[20];

    void addSongs()  // input songs
    {
        cout<<"Enter first song: ";
        cin>>song1;
        cout<<"Enter second song: ";
        cin>>song2;
    }

    void removeSong2()  // remove second song
    {
        song2[0] = '\0';
    }

    void playRandom()  // pick any one song
    {
        if(song2[0] != '\0')
            cout<<"Playing: "<<song1<<endl;  // pick first song
        else
            cout<<"Playing: "<<song1<<endl;  // only first song available
    }

    void display()  // show songs
    {
        cout<<"Songs:\n"<<song1<<" "<<song2<<endl;
    }
};

int main()
{
    MusicLibrary m;

    m.addSongs();

    cout<<"\nBefore removing:\n";
    m.display();

    m.removeSong2();

    cout<<"\nAfter removing:\n";
    m.display();

    cout<<"Random song: ";
    m.playRandom();

    return 0;
}
