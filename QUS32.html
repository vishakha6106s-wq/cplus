Enter an array from the user and print the number which is positive?


#include<iostream>
using namespace std;
int main()
{
int a[20],n,i;
cout<<"Enter the size of an array:";
cin>>n;
cout<<"Enter the elements:"<<endl;
for(i=0;i<n;i++)
{
cin>>a[i];
}
cout<<"Positive numbers"<<endl;
for(i=0;i<n;i++)
{
if(a[i]>=0)
{
cout<<a[i]<<endl;
}
}
return 0;
}
