Program to Convert Enum to String


#include<iostream>
using namespace std;

enum Day { MON, TUE, WED };

int main()
{
    int d;

    cout<<"Enter number (0-2): ";
    cin>>d;

    if(d==0)
        cout<<"MONDAY";
    else if(d==1)
        cout<<"TUESDAY";
    else if(d==2)
        cout<<"WEDNESDAY";

    return 0;
}

Output


Enter number (0-2): 2
WEDNESDAY
