Write a C++ program to create a class called "Airplane" with a flight number, destination, and departure time attributes, and methods to check flight status and delay.


#include<iostream>
using namespace std;

class Airplane
{
public:
    char flightNo[20];
    char destination[20];
    int time;

    void checkStatus()
    {
        cout<<"Flight is On Time"<<endl;
    }

    void delay()
    {
        time = time + 1;   // delay by 1 hour
    }

    void display()
    {
        cout<<"Flight No: "<<flightNo<<endl;
        cout<<"Destination: "<<destination<<endl;
        cout<<"Time: "<<time<<endl;
    }
};

int main()
{
    Airplane a;

    // input
    cout<<"Enter flight no:";
    cin>>a.flightNo;
    cout<<"Enter the destination:";
    cin>>a.destination;
    cout<<"Enter the time:";
    cin>>a.time;

    // show status
    a.checkStatus();

    a.delay();

    cout<<"\nAfter delay:\n";
    a.display();

    return 0;
}
