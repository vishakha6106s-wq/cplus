Define a class Complex to represent complex numbers. Overload the + operator to add two complex numbers.


#include <iostream>
using namespace std;
class Complex {
    int r, i;
public:
    Complex(int x = 0, int y = 0) {
        r = x;
        i = y;
    }
    Complex operator+(Complex c) {
        return Complex(r + c.r, i + c.i);
    }
    void show() {
        cout << r << " + " << i << "i";
    }
};
int main() {
    Complex a(2, 3), b(4, 5), c;
    c = a + b;
    c.show();
    return 0;
}

Output


6+8i
