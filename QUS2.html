Program to find the frequency of each elemnt in the array.


#include<iostream>
using namespace std;
int main()
{
int arr[50],n,i,j,count;
cout<<"Enter number of elements:";
cin>>n;
cout<<"Enter elements:";
for(i=0;i<n;i++)
{
cin>>arr[i];
}
for(i=0;i<n;i++)
{
count= 1;
if(arr[i]== -1)
{
continue;
}
for(j=i+1;j<n;j++)
{
if(arr[i]==arr[j])
{
count++;
arr[j]= -1;
}
}
cout<<arr[i]<<" occurs "<<count<<" times "<<endl;
}
return 0;
}
