Create a class entering the rollno, name and class of the student from user but rollno should be automatically generated as we enter the information of 10 students


#include <iostream>
using namespace std;

class Student {
private:
    int rollNo;
    string name;
    string studentClass;
    static int count; // Automatic roll numbers

public:
    Student() {
        count++;
        rollNo = count;

        cout << "Entering details of Student " << rollNo << ":\n";

        cout << "Enter Name: ";
        cin >> name;

        cout << "Enter Class: ";
        cin >> studentClass;
    }
    void display() 
    {
        cout << "Roll No: " << rollNo << ", Name: " << name << ", Class: " << studentClass << endl;
    }
};

int Student::count = 0;

int main() {
    Student s[10]; // 10 students
    cout << "\nStudent Details:\n";
    for (int i = 0; i < 10; i++) {
        s[i].display(); // Display each student details separately
    }

    return 0;
}
